server {
    listen 80;
    listen [::]:80;
    server_name futeur.app www.futeur.app;

    location / {
        return 301 https://$host$request_uri;
    }
}

# HTTPS server block to proxy requests to your app
server {
    listen 443 ssl;
    listen [::]:443 ssl;
    server_name futeur.app www.futeur.app;

    ssl_certificate /etc/letsencrypt/live/futeur.app/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/futeur.app/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;
    ssl_ciphers HIGH:!aNULL:!MD5;

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}
